1. Create `__tests__/firestore-rules/setup.ts` — shared test environment factory:
   ```typescript
   import { initializeTestEnvironment, RulesTestEnvironment } from '@firebase/rules-unit-testing';
   import { doc, setDoc, Timestamp } from 'firebase/firestore';
   import { readFileSync } from 'fs';
   import { resolve } from 'path';

   let testEnv: RulesTestEnvironment;

   export async function setupTestEnvironment(): Promise<RulesTestEnvironment> {
     testEnv = await initializeTestEnvironment({
       projectId: 'towlink-test-' + Date.now(),
       firestore: {
         rules: readFileSync(resolve(__dirname, '../../firestore.rules'), 'utf8'),
         host: 'localhost',
         port: 8080,
       },
     });
     return testEnv;
   }

   export async function seedUser(
     env: RulesTestEnvironment,
     uid: string,
     role: string,
     extra?: Record<string, unknown>
   ) {
     await env.withSecurityRulesDisabled(async (ctx) => {
       await setDoc(doc(ctx.firestore(), 'users', uid), {
         email: `${uid}@test.com`,
         role,
         createdAt: Timestamp.now(),
         ...extra,
       });
     });
   }

   export async function seedCompany(
     env: RulesTestEnvironment,
     companyId: string,
     ownerUid: string
   ) {
     await env.withSecurityRulesDisabled(async (ctx) => {
       await setDoc(doc(ctx.firestore(), 'companies', companyId), {
         name: 'Test Tow Yard',
         ownerUid,
         address: '123 Test St',
         createdAt: Timestamp.now(),
       });
     });
   }
   ```

2. Create `__tests__/firestore-rules/requests.rules.test.ts` — test cases:
   - ALLOW: commuter creates request with all required fields, commuterId == auth.uid, status == 'searching'
   - DENY: driver creates request (isCommuter() returns false for driver role)
   - DENY: unauthenticated user creates request
   - ALLOW: commuter cancels own request (status -> 'cancelled')
   - DENY: commuter cancels another commuter's request
   - ALLOW: driver claims a searching request (status 'searching' -> 'claimed', claimedByDriverId == auth.uid)
   - ALLOW: driver accepts a claimed request they claimed (status 'claimed' -> 'accepted')
   - DENY: driver accepts a request claimed by another driver
   - ALLOW: driver declines a claimed request (status -> 'searching', claimedByDriverId -> null)
   - ALLOW: any authenticated user reads requests

   The valid request data object for creates must include ALL required fields from the rules:
   `commuterId, location ({latitude, longitude}), dropoffLocation ({latitude, longitude}), pickupAddress, dropoffAddress, vehicleInfo ({year: number, make: string, model: string}), estimatedPrice (positive number), totalTripDistance (positive number), serviceType: 'tow', status: 'searching', createdAt (Timestamp), expiresAt (Timestamp)`.

   CRITICAL: Every test must call `seedUser()` FIRST to create the user doc with the correct role. Without this, isCommuter()/isDriver() always returns false (see Pitfall 5 in RESEARCH.md).
