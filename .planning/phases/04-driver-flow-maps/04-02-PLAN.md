```typescript
      if (cancelled) subscription.remove();

  })();

  return () => {

    subscription?.remove();
  };
```

**GPS-to-step matching useEffect** (per D-04, UI-SPEC thresholds: 30m advance, 150m reroute):
```typescript
useEffect(() => {
  if (!driverLocation || !routeData || !routeData.steps.length) return;
  if (!trip || trip.status === 'completed' || trip.status === 'cancelled') return;

  const steps = routeData.steps;
  if (currentStepIndex >= steps.length - 1) return;

  const nextStep = steps[currentStepIndex + 1];
  const distToNextKm = getDistanceInKm(driverLocation, nextStep.startLocation);

  if (distToNextKm * 1000 < 30) {
    setCurrentStepIndex((prev) => prev + 1);
    return;
  }

  const currentStep = steps[currentStepIndex];
  const distToCurrentKm = getDistanceInKm(driverLocation, currentStep.startLocation);

  if (distToCurrentKm * 1000 > 150) {
    // Off route -- reroute per D-05
    const destination = trip.status === 'in_progress'
      ? trip.dropoffLocation
      : trip.pickupLocation;
    fetchDirections(driverLocation, destination).then((result) => {
      if (result) {
        setRouteData(result);
        setCurrentStepIndex(0);
      }
    });
  }
}, [driverLocation, routeData, currentStepIndex]);
```

```typescript
const handleOpenInMaps = async () => {
  if (!trip) return;
  const coords = trip.pickupLocation;
  const googleMapsUrl = `comgooglemaps://?daddr=${coords.latitude},${coords.longitude}`;
  const appleMapsUrl = `maps://?daddr=${coords.latitude},${coords.longitude}`;
  const googleWebUrl = `https://www.google.com/maps/dir/?api=1&destination=${coords.latitude},${coords.longitude}`;

  const canOpenGoogle = await Linking.canOpenURL(googleMapsUrl);
  if (canOpenGoogle) {
    await Linking.openURL(googleMapsUrl);
  } else {
    const canOpenApple = await Linking.canOpenURL(appleMapsUrl);
    await Linking.openURL(canOpenApple ? appleMapsUrl : googleWebUrl);
  }
};
```
